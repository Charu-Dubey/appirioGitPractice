public class MyRelatedNotesController {
    public Id userId {get; set;}
    public String noteType {get; set;}
    public String selectedRecId {get; set;}
    public Boolean displayAction {get; set;}
    public Set<Id> accId;
    public Set<Id> oppId;
    public Set<Id> caseId;
    public List<Note__c> notelist {get; set;}
    public List<SelectOption> recList {get; set;}
    public MyRelatedNotesController()
    {
        recList = new List<SelectOption>();
        noteList = new List<Note__c>();
        displayAction = false;
        accId = new Set<Id>();
        oppId = new Set<Id>();
        caseId = new Set<Id>();
        for(Note__c note : [SELECT Id, Account__c, Opportunity__c, Case__c
                            FROM Note__c
                            WHERE Account__c!= NULL
                            OR Opportunity__c!= NULL
                            OR Case__c!= NULL])
        {
            if(note.Account__c != NULL)
            {
                accId.add(note.Account__c);
            }
            if(note.Opportunity__c != NULL)
            {
                oppId.add(note.Opportunity__c);
            }
            if(note.Case__c != NULL)
            {
                caseId.add(note.Case__c);
            }
        }
    }
    public void process()
    {
        if(userId == NULL)
        {
            userId = UserInfo.getUserId();
        }
        if(noteType == NULL)
        {
            noteType = 'Opportunity';
        }
        if(noteType.capitalize() == 'Account')
        {
            system.debug('in if loop');
            for(Account acc : [SELECT Id, Name
                               FROM Account
                               WHERE Id IN : accId])
            {
                system.debug('in for loop');
                recList.add(new SelectOption(acc.Id, acc.Name));
            }
            system.debug(recList);
        }
        if(noteType.capitalize() == 'Opportunity')
        {
            for(Opportunity opp : [SELECT Id, Name
                                   FROM Opportunity
                                   WHERE Id IN : oppId])
            {
                recList.add(new SelectOption(opp.Id, opp.Name));
            }
        }
        if(noteType.capitalize() == 'Case')
        {
            for(Case caseRec : [SELECT Id, CaseNumber
                                FROM Case
                                WHERE Id IN : caseId])
            {
                recList.add(new SelectOption(caseRec.Id, caseRec.CaseNumber));
            }
        }
    }
    public void getDetails()
    {
        if(noteType.capitalize() == 'Account')
        {
            for(Note__c note : [SELECT Id, Name, Date__c, Description__c, OwnerId
                                FROM Note__c
                                WHERE Account__c =: selectedRecId])
            {
                noteList.add(note);
                if(note.OwnerId == userId)
                {
                    displayAction = true;
                }
            }
        }
        if(noteType.capitalize() == 'Opportunity')
        {
            for(Note__c note : [SELECT Id, Name, Date__c, Description__c, OwnerId
                                FROM Note__c
                                WHERE Opportunity__c =: selectedRecId])
            {
                noteList.add(note);
                if(note.OwnerId == userId)
                {
                    displayAction = true;
                }
            }
        }
        if(noteType.capitalize() == 'Case')
        {
            for(Note__c note : [SELECT Id, Name, Date__c, Description__c, OwnerId
                                FROM Note__c
                                WHERE Case__c =: selectedRecId])
            {
                noteList.add(note);
                if(note.OwnerId == userId)
                {
                    displayAction = true;
                }
            }
        }
    }

    {
        
    }
}
